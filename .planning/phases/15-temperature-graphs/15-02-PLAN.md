### Task 1: Create firing_chart.js
Create `static/js/firing_chart.js` — under 60 lines of vanilla JS:

```js
// firing_chart.js — initialise Chart.js temperature curve from /api/firings/{id}/readings
(function () {
  const canvas = document.getElementById('firingChart');
  if (!canvas) return;

  const firingId = canvas.dataset.firingId;

  fetch('/api/firings/' + firingId + '/readings')
    .then(function (res) { return res.json(); })
    .then(function (data) {
      const readings = data.readings || [];
      const labels = readings.map(function (r) { return r.elapsed_minutes; });
      const temps  = readings.map(function (r) { return r.temperature; });

      // Point labels: gas + flue setting per point
      const pointLabels = readings.map(function (r) {
        return (r.gas_setting || '') + (r.flue_setting ? ' / ' + r.flue_setting : '');
      });

      new Chart(canvas, {
        type: 'line',
        data: {
          labels: labels,
          datasets: [{
            label: 'Temperature (°C)',
            data: temps,
            borderColor: '#c0392b',
            backgroundColor: 'rgba(192,57,43,0.1)',
            pointRadius: 4,
            tension: 0.2,
          }]
        },
        options: {
          responsive: true,
          plugins: {
            tooltip: {
              callbacks: {
                afterLabel: function (ctx) {
                  return pointLabels[ctx.dataIndex] || '';
                }
              }
            }
          },
          scales: {
            x: { title: { display: true, text: 'Elapsed (min)' } },
            y: { title: { display: true, text: 'Temperature (°C)' } }
          }
        }
      });
    })
    .catch(function (err) { console.error('firing chart error', err); });
}());
```
